\section*{Phase 5 : Proof Trees}

\bigskip
\noindent
\textbf{Example 1 : Simple Modus Ponens}

\begin{itemize}
\item $p \Rightarrow q$ [premise]
\item $p$ [premise]
\begin{itemize}
\item $q$ [modus ponens]
\end{itemize}
\end{itemize}

\bigskip
\noindent
\textbf{Example 2 : Conjunction Elimination}

\begin{itemize}
\item $p \land q$ [premise]
\begin{itemize}
\item $p$ [$\land$ elim left]
\item $q$ [$\land$ elim right]
\end{itemize}
\end{itemize}

\bigskip
\noindent
\textbf{Example 3 : Nested Proof}

\begin{itemize}
\item $(p \Rightarrow q) \land (q \Rightarrow r)$ [premise]
\begin{itemize}
\item $p \Rightarrow q$ [$\land$ elim]
\item $q \Rightarrow r$ [$\land$ elim]
\end{itemize}
\item $p$ [premise]
\begin{itemize}
\item $q$ [modus ponens]
\begin{itemize}
\item $r$ [modus ponens]
\end{itemize}
\end{itemize}
\end{itemize}

\bigskip
\noindent
\textbf{Example 4 : Disjunction Cases}

\begin{itemize}
\item $p \lor q$ [premise]
\begin{itemize}
\item $p$ [case 1]
\begin{itemize}
\item $p \Rightarrow r$ [premise]
\begin{itemize}
\item $r$ [modus ponens]
\end{itemize}
\end{itemize}
\item $q$ [case 2]
\begin{itemize}
\item $q \Rightarrow r$ [premise]
\begin{itemize}
\item $r$ [modus ponens]
\end{itemize}
\end{itemize}
\end{itemize}
\end{itemize}

\bigskip
\noindent
\textbf{Example 5 : Proof with Quantifiers}

\begin{itemize}
\item $\forall x \colon N \bullet x > 0$ [premise]
\item $a \in N$ [premise]
\begin{itemize}
\item $a > 0$ [universal instantiation]
\end{itemize}
\end{itemize}
