\section*{Schema Calculus: Composition}
\addcontentsline{toc}{section}{Schema Calculus: Composition}

\subsection*{Example 1 : Sequential operations}
\addcontentsline{toc}{subsection}{Example 1 : Sequential operations}

\begin{schema}{OpA}
x : \nat \\
x' : \nat
\where
x' > x
\end{schema}

\begin{schema}{OpB}
x : \nat \\
x' : \nat
\where
x' < 100
\end{schema}

\begin{zed}
OpAB \defs OpA \semi OpB
\end{zed}

\subsection*{Example 2 : Three - way composition}
\addcontentsline{toc}{subsection}{Example 2 : Three - way composition}

\begin{schema}{Inc}
n : \nat \\
n' : \nat
\where
n' = n + 1
\end{schema}

\begin{schema}{Double}
n : \nat \\
n' : \nat
\where
n' = n + n
\end{schema}

\begin{schema}{Clamp}
n : \nat \\
n' : \nat
\where
n' = 100
\end{schema}

\begin{zed}
IncThenDouble \defs Inc \semi Double
\end{zed}

\begin{zed}
IncThenDoubleClamp \defs Inc \semi Double \semi Clamp
\end{zed}
