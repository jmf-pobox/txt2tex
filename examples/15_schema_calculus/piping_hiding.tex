\subsection*{Example 1 : Schema piping ( output - to - input pipeline )}
\addcontentsline{toc}{subsection}{Example 1 : Schema piping ( output - to - input pipeline )}

\begin{schema}{Send}
chan! : \nat
\where
chan! > 0
\end{schema}

\begin{schema}{Receive}
chan? : \nat
\where
chan? < 100
\end{schema}

\begin{zed}
Pipeline \defs Send \pipe Receive
\end{zed}

\subsection*{Example 2 : Schema hiding}
\addcontentsline{toc}{subsection}{Example 2 : Schema hiding}

\begin{schema}{State}
x : \nat \\
y : \nat \\
temp : \nat
\where
x < y \\
temp > 0
\end{schema}

\begin{zed}
Visible \defs State \hide (temp)
\end{zed}

\subsection*{Example 3 : Combined piping and hiding}
\addcontentsline{toc}{subsection}{Example 3 : Combined piping and hiding}

\begin{schema}{OpA}
x : \nat \\
x' : \nat
\where
x' > x
\end{schema}

\begin{schema}{OpB}
x : \nat \\
x' : \nat
\where
x' < 100
\end{schema}

\begin{zed}
OpFiltered \defs OpA \semi OpB \hide (x)
\end{zed}
