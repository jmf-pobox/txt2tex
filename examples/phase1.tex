\section*{Propositional Logic}

\bigskip
\noindent
\textbf{Solution 1}

\medskip

\noindent
\hangindent=2em
(a) $p \land q \Rightarrow p$

\medskip
(b)
\par
\vspace{11pt}
\begin{tabular}{c|c|c}
$p$ & $q$ & $\mathbf{p \Rightarrow q}$ \\
\hline
\textit{t} & \textit{t} & \textbf{t} \\
\textit{t} & \textit{f} & \textbf{f} \\
\textit{f} & \textit{t} & \textbf{t} \\
\textit{f} & \textit{f} & \textbf{t} \\
\end{tabular}

\medskip

\noindent
\hangindent=2em
(c) $\lnot (p \land q) \Leftrightarrow \lnot p \lor \lnot q$

\textbf{Solution 2}

\medskip

\noindent
\hangindent=2em
(a) $(p \lor q) \land r \Leftrightarrow p \land r \lor q \land r$

\medskip
(b)
\par
\vspace{11pt}
\begin{tabular}{c|c|c}
$p$ & $q$ & $\mathbf{p \Leftrightarrow q}$ \\
\hline
\textit{t} & \textit{t} & \textbf{t} \\
\textit{t} & \textit{f} & \textbf{f} \\
\textit{f} & \textit{t} & \textbf{f} \\
\textit{f} & \textit{f} & \textbf{t} \\
\end{tabular}

\medskip

\noindent
\hangindent=2em
(c) $p \Rightarrow (q \Rightarrow r)$
