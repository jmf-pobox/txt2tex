\section*{Propositional Logic}

\bigskip
\noindent
\textbf{Solution 1}

\medskip

(a)
\par
\vspace{11pt}
$p \land q \Rightarrow p$

\medskip

(b)
\par
\vspace{11pt}
\begin{center}
\begin{tabular}{c c c}
$p$ & $q$ & $p \Rightarrow q$ \\
\hline
t & t & t \\
t & f & f \\
f & t & t \\
f & f & t \\
\end{tabular}
\end{center}

\medskip

(c)
\par
\vspace{11pt}
$\lnot (p \land q) \Leftrightarrow \lnot p \lor \lnot q$

\medskip

\bigskip
\noindent
\textbf{Solution 2}

\medskip

(a)
\par
\vspace{11pt}
$(p \lor q) \land r \Leftrightarrow p \land r \lor q \land r$

\medskip

(b)
\par
\vspace{11pt}
\begin{center}
\begin{tabular}{c c c}
$p$ & $q$ & $p \Leftrightarrow q$ \\
\hline
t & t & t \\
t & f & f \\
f & t & f \\
f & f & t \\
\end{tabular}
\end{center}

\medskip

(c)
\par
\vspace{11pt}
$(p \Rightarrow q) \Rightarrow r$

\medskip
