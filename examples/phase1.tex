\bigskip
\noindent
\textbf{Solution 2}

(a)
$p \lor q \land r \Leftrightarrow p \land r \lor q \land r$

\medskip

(b)
\begin{center}
\begin{tabular}{|c|c|c|}
$p$ & $q$ & $p <=> q$ \\
\hline
T & T & T \\
T & F & F \\
F & T & F \\
F & F & T \\
\end{tabular}
\end{center}

\medskip

(c)
$p \Rightarrow q \Rightarrow r$
