\section*{Phase 11 . 9 : Generic Type Instantiation}

\noindent
\textbf{Example 1 : Basic Generic Instantiation}

\medskip

(a)
\par
\vspace{11pt}

\medskip

(b)
\par
\vspace{11pt}
$seq[N]$

\medskip

(c)
\par
\vspace{11pt}
$P[X]$

\medskip
