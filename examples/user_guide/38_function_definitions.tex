\begin{axdef}
square : \nat \fun \nat
\where
\forall n : \nat @ square(n) = n * n
\end{axdef}

\begin{axdef}
predecessor : \nat \pfun \nat
\where
\forall n : \nat @ n > 0 \implies predecessor(n) = n - 1
\end{axdef}

\begin{gendef}[X]
  identity: X \fun X
\where
  \forall x : X @ identity(x) = x
\end{gendef}

\begin{gendef}[X, Y]
  proj1: X \cross Y \fun X
\where
  \forall x : X @ \forall y : Y @ proj1(x, y) = x
\end{gendef}
