// Example: Axdef - Global Scope
// Demonstrates axdef blocks with global identifiers
// From USER_GUIDE.md - Schema Notation section

zed
  given ShowId, EpisodeId
end


axdef
  shows : F ShowId
  show_episodes : ShowId +-> F EpisodeId
where
  shows subset dom show_episodes
end
