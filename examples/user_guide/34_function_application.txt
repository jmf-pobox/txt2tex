// Example: Function Application
// Demonstrates various forms of function application
// From USER_GUIDE.md - Functions section

given X, Y, A, Entry, Person

axdef
  f : X -> Y
  g : Y -> Y
  h : Y -> Y
  R : X <-> Y
  S : Y <-> Y
where
  f(x) = x
  g(x, y, z) = x
  seq(Entry) = {}
  P(Person) = {}
  f(g(h(x))) = x
end

(f ++ g)(x)

<a, b, c>(2)

(R o9 S)(| A |)
