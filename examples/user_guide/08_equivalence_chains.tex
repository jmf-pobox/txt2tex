      \lnot (p \land q) \\
      \Leftrightarrow \lnot p \lor \lnot q & [\mbox{De Morgan}] \\
      \Leftrightarrow \lnot p \lor \lnot q & [\mbox{idempotence}] \\
      \Leftrightarrow \lnot p \lor \lnot q \lor \lnot r &
      [\mbox{associativity}] \\
      \Leftrightarrow true & [\mbox{excluded middle}]
    \end{array}$%
  }
