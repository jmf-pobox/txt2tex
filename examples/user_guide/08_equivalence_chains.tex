not(p(and)(q)) \\
\Leftrightarrow not(p)(or(not(q))) & [\mbox{De Morgan}] \\
\Leftrightarrow not(p)(or(not(q))) & [\mbox{idempotence}] \\
\Leftrightarrow not(p)(or(not(q)))(or(not(r))) & [\mbox{associativity}] \\
\Leftrightarrow true & [\mbox{excluded middle}]
\end{array}$%
}
