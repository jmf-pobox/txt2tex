\begin{axdef}
  successor : \nat \fun \nat \\
  square : \nat \fun \nat
  \where
  \forall n : \nat @ successor(n) = n + 1 \land square(n) = n * n
\end{axdef}

\begin{zed}
  squareSuccessor == square \circ successor
  \also
  result == square(successor(5))
