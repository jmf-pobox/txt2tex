// Example: Free Types
// Demonstrates basic free type definitions
// From USER_GUIDE.md - Definitions section

zed
  IntTree ::= leaf | node<N cross IntTree cross IntTree>
end


axdef
  emptyTree : IntTree
  treeDepth : IntTree -> N
where
  emptyTree = leaf
  treeDepth(leaf) = 0
end
