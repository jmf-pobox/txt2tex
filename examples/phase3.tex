\section*{Phase 3 : Quantifiers and Mathematical Notation}

\bigskip
\noindent
\textbf{Example 1 : Quantifiers}

\medskip

(a)
\par
\vspace{11pt}
$\forall x \colon N \bullet x \geq 0$

\medskip

(b)
\par
\vspace{11pt}
$\exists y \colon N \bullet y = 0$

\medskip

(c)
\par
\vspace{11pt}
$\forall n \colon N \bullet n \geq 0 \land n < 100$

\medskip

\bigskip
\noindent
\textbf{Example 2 : Subscripts and Superscripts}

\medskip

(a)
\par
\vspace{11pt}
$x^2$

\medskip

(b)
\par
\vspace{11pt}
$a_1$

\medskip

(c)
\par
\vspace{11pt}
$x_i^2$

\medskip

\bigskip
\noindent
\textbf{Example 3 : Comparison Operators}

\medskip

(a)
\par
\vspace{11pt}
$x < 10$

\medskip

(b)
\par
\vspace{11pt}
$x \geq 0 \land x \leq 100$

\medskip

(c)
\par
\vspace{11pt}
$y > 5 \lor y = 0$

\medskip

\bigskip
\noindent
\textbf{Example 4 : Set Operators}

\medskip

(a)
\par
\vspace{11pt}
$x \in N$

\medskip

(b)
\par
\vspace{11pt}
$A \subseteq B$

\medskip

(c)
\par
\vspace{11pt}
$A \cup B \cap C$

\medskip

\bigskip
\noindent
\textbf{Example 5 : Complex Expressions}

\medskip

(a)
\par
\vspace{11pt}
\vspace{-10pt}
\begin{align*}
\forall x \colon N \bullet x \in N \land x \geq 0 \\
&\Leftrightarrow \forall x \colon N \bullet x \geq 0
\end{align*}

\medskip

(b)
\par
\vspace{11pt}
$\forall i \colon N \bullet a_i > 0 \Rightarrow a_i^2 > 0$

\medskip

(c)
\par
\vspace{11pt}
\begin{center}
\begin{tabular}{c c c}
$p$ & $q$ & $p \Rightarrow q$ \\
\hline
t & t & t \\
t & f & f \\
f & t & t \\
f & f & t \\
\end{tabular}
\end{center}
