\section*{Phase 3 : Quantifiers and Mathematical Notation}

\bigskip
\noindent
\textbf{Example 1 : Quantifiers}

(a)
$\forall x \colon N \bullet x \geq 0$

\medskip

(b)
$\exists y \colon N \bullet y = 0$

\medskip

(c)
$\forall n \colon N \bullet n \geq 0 \land n < 100$

\medskip

\bigskip
\noindent
\textbf{Example 2 : Subscripts and Superscripts}

(a)
$x^2$

\medskip

(b)
$a_1$

\medskip

(c)
$x_i^2$

\medskip

\bigskip
\noindent
\textbf{Example 3 : Comparison Operators}

(a)
$x < 10$

\medskip

(b)
$x \geq 0 \land x \leq 100$

\medskip

(c)
$y > 5 \lor y = 0$

\medskip

\bigskip
\noindent
\textbf{Example 4 : Set Operators}

(a)
$x \in N$

\medskip

(b)
$A \subseteq B$

\medskip

(c)
$A \cup B \cap C$

\medskip

\bigskip
\noindent
\textbf{Example 5 : Complex Expressions}

(a)
\begin{align*}
\forall x \colon N \bullet x \in N \land x \geq 0 \\
&\Leftrightarrow \forall x \colon N \bullet x \geq 0
\end{align*}

\medskip

(b)
$\forall i \colon N \bullet a_i > 0 \Rightarrow a_i^2 > 0$

\medskip

(c)
\begin{center}
\begin{tabular}{|c|c|c|}
$p$ & $q$ & $p => q$ \\
\hline
T & T & T \\
T & F & F \\
F & T & T \\
F & F & T \\
\end{tabular}
\end{center}
