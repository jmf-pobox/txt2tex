** Example: Semicolon bindings in axdef **

axdef
  dummy : N
where
  dummy = 0
  forall x : N; y : N | x + y > 0
end

** Example: Function on sequences **

given Entry

axdef
  seqlen : seq Entry -> N
where
  seqlen (<>) = 0
  forall x : Entry; s : seq Entry | seqlen (<x> ^ s) = 1 + seqlen (s)
end

** Example: Nested quantifier **

given Person

axdef
  constraint : N
where
  constraint = 1
  forall p : Person; n : N | n > 0
end
