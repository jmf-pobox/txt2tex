\section*{Phase 18 Demo : Partial Function Operators}

\section*{Example 1 : Partial injection operator}
\addcontentsline{toc}{section}{Example 1 : Partial injection operator}

\noindent Basic function type with partial injection ($\pinj$) operator:

\bigskip

\begin{axdef}
  f : \nat \pinj \nat
  \where
  \dom f = \{1, 2, 3\} \\
  f(1) = 10 \\
  f(2) = 20 \\
  f(3) = 30
\end{axdef}

\section*{Example 2 : Partial function operator}
\addcontentsline{toc}{section}{Example 2 : Partial function operator}

\noindent Basic function type with partial function ($\pfun$) operator:

\bigskip

\begin{axdef}
  g : \nat \pfun \nat
  \where
  \dom g = \{1, 2\} \\
  g(1) = 100 \\
  g(2) = 200
\end{axdef}

\section*{Example 3 : Complex types with partial injection}
\addcontentsline{toc}{section}{Example 3 : Complex types with partial injection}

\noindent Using partial injection ($\pinj$) with custom types:

\bigskip

\begin{zed}
  Year == \nat
\end{zed}

\begin{zed}
  [Entry]
\end{zed}

\begin{zed}
  Table == \seq~Entry
\end{zed}

\begin{axdef}
  records : Year \pinj Table
  \where
  \dom records = \{1993, 1994, 1995\}
\end{axdef}
