\section*{Phase 19 Demo : Finite Set Types}

\section*{Example 1 : F ( finite sets )}
\addcontentsline{toc}{section}{Example 1 : F ( finite sets )}

\noindent F X denotes all finite subsets of X:

\bigskip

\begin{zed}
[SongId, UserId]
\end{zed}

\begin{axdef}
songs : \finset SongId \\
users : \finset UserId
\end{axdef}

\section*{Example 2 : F1 ( non - empty finite sets )}
\addcontentsline{toc}{section}{Example 2 : F1 ( non - empty finite sets )}

\noindent F1 X denotes all non-empty finite subsets of X:

\bigskip

\begin{axdef}
activeUsers : \finset_1 UserId
\end{axdef}

\section*{Example 3 : Combined with function types}
\addcontentsline{toc}{section}{Example 3 : Combined with function types}

\noindent Using F and F1 with partial functions:

\bigskip

\begin{zed}
[PlaylistId, Playlist]
\end{zed}

\begin{axdef}
playlists : PlaylistId \pfun Playlist \\
playlistOwner : PlaylistId \pfun UserId \\
playlistSubscribers : PlaylistId \pfun \finset_1 UserId
\end{axdef}
