\section*{Test : Union in Quantifier Domain}

\begin{zed}[~ UserId, SongId ~]
\end{zed}

\begin{axdef}
  users : \finset UserId \\
  songs : \finset SongId \\
  hated : UserId \pfun \finset SongId \\
  loved : UserId \pfun \finset SongId
  \where
  \forall i : \dom hated \cup \dom loved @ hated(i) \cap loved(i) = \emptyset
\end{axdef}
