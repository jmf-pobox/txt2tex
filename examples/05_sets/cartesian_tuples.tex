\section*{Phase 8 : Set Comprehension}

\section*{Example 1 : Set by Predicate}
\addcontentsline{toc}{section}{Example 1 : Set by Predicate}

\subsection*{(a)}

\noindent Basic set comprehension with domain

\bigskip

\noindent
$\{~ x : \nat | x > 0 ~\}$


\medskip

\subsection*{(b)}

\noindent Multi variable set

\bigskip

\noindent
$\{~ x, y : \nat | x = y ~\}$


\medskip

\subsection*{(c)}

\noindent Set without domain

\bigskip

\noindent
$\{~ x | x \in A ~\}$


\section*{Example 2 : Set by Expression}
\addcontentsline{toc}{section}{Example 2 : Set by Expression}

\subsection*{(a)}

\noindent Set comprehension with expression part

\bigskip

\noindent
$\{~ x : \nat | x > 0 @ x \bsup 2 \esup ~\}$


\medskip

\subsection*{(b)}

\noindent Combining set comprehension with quantifier

\bigskip

\noindent
$\forall x : \nat @ x \in \{~ y : \nat | y > 0 ~\}$


\section*{Example 3 : Complex Examples}
\addcontentsline{toc}{section}{Example 3 : Complex Examples}

\subsection*{(a)}

\noindent Set comprehension in equivalence

\bigskip

\noindent
$\{~ x : \nat | x > 0 ~\} = \{~ y : \nat | y > 0 ~\}$


\medskip

\subsection*{(b)}

\noindent Set comprehension with multiple conditions

\bigskip

\noindent
$\{~ x : \nat | x > 0 \land x \leq 10 ~\}$
