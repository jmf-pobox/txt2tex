\section*{Testing Anonymous Schemas}

\bigskip
\noindent
\textbf{Test 1 : Named Schema}

\medskip

\begin{schema}{State}
count : N
\where
count \geq 0
\end{schema}

\bigskip
\noindent
\textbf{Test 2 : Anonymous Schema}

\medskip

\begin{schema}{}
hd : seq[N]
\where
\# hd \leq 100
\end{schema}

\bigskip
\noindent
\textbf{Test 3 : Another Anonymous Schema}

\medskip

\begin{schema}{}
x : N
y : N
\where
x + y < 100
\end{schema}
