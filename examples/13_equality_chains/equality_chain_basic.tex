\subsection*{Example 1 : Base case - length of the empty sequence}
\addcontentsline{toc}{subsection}{Example 1 : Base case - length of the empty sequence}

\subsubsection*{(a)}
\addcontentsline{toc}{subsubsection}{(a)}

\noindent Prove length $\langle \rangle$ = 0.

\bigskip

\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\begin{array}{l@{\hspace{2em}}r}
length(\langle \rangle) \\
= 0 & [\mbox{by definition of length}]
\end{array}$%
}
\end{center}
\bigskip

\subsection*{Example 2 : Inductive step - length of a non - empty sequence}
\addcontentsline{toc}{subsection}{Example 2 : Inductive step - length of a non - empty sequence}

\subsubsection*{(a)}
\addcontentsline{toc}{subsubsection}{(a)}

\noindent Prove length (s \textasciicircum$\{\}$ $\langle x \rangle$) = length s + 1.

\bigskip

\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\begin{array}{l@{\hspace{2em}}r}
length(s \cat \langle x \rangle) \\
= length(s) + 1 & [\mbox{by definition of length and concatenation}]
\end{array}$%
}
\end{center}
\bigskip

\subsection*{Example 3 : Multi - step chain with justifications}
\addcontentsline{toc}{subsection}{Example 3 : Multi - step chain with justifications}

\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\begin{array}{l@{\hspace{2em}}r}
length(s \cat t) \\
= length(s) + length(t) & [\mbox{by length - concatenation lemma}] \\
= length(t) + length(s) & [\mbox{commutativity of +}]
\end{array}$%
}
\end{center}
