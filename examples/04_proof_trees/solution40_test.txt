** Example 1: Simple recursive function **

axdef
  length : seq EntryS40 -> N
where
  length(<>) = 0
  forall x : EntryS40; s : seq EntryS40 | length(<x> ^ s) = 1 + length(s)
end

** Example 2: Identity function on sequences **

axdef
  identityS40 : seq EntryS40 -> seq EntryS40
where
  identityS40(<>) = <>
  forall x : EntryS40; s : seq EntryS40 | identityS40(<x> ^ s) = <x> ^ identityS40(s)
end

** Example 3: Sequence doubling **

axdef
  duplicate : seq N -> seq N
where
  duplicate(<>) = <>
  forall n : N; s : seq N | duplicate(<n> ^ s) = <n, n> ^ duplicate(s)
end
