\begin{zed}
[EntryPM]
\end{zed}

\begin{axdef}
filterSeq : \seq~EntryPM \fun \seq~EntryPM
\where
filterSeq(\langle \rangle) = \langle \rangle \\
\forall x : EntryPM @ \forall s : \seq~EntryPM @ filterSeq(\langle x \rangle \cat s) = \langle x \rangle \cat filterSeq(s)
\end{axdef}

\noindent This demonstrates pattern matching on sequences with base case ($\langle \rangle$) and recursive case ($\langle x \rangle$ \textasciicircum$\{\}$ s).
