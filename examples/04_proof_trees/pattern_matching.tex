\section*{Test Pattern Matching}
\addcontentsline{toc}{section}{Test Pattern Matching}

\begin{zed}
  [Entry]
\end{zed}

\begin{axdef}
  filterSeq : \seq Entry \fun \seq Entry
  \where
  filterSeq(\langle \rangle) = \langle \rangle \\
  \forall x : Entry @ \forall s : \seq Entry @ filterSeq(\langle x
  \rangle \cat s) = \langle x \rangle \cat filterSeq(s)
\end{axdef}

\noindent This demonstrates pattern matching on sequences with base
case ($\langle \rangle$) and recursive case ($\langle x \rangle$ ^ s).
