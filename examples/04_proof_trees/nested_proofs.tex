\section*{Proof Tree Nesting Depth Test}

\bigskip
\noindent
\textbf{Test 1 : Simple ( depth 1 )}

\medskip

\noindent
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \implies q}{
  \ulcorner p \urcorner^{[1]}
&
\infer[axiom]{q}{}
}
$

\bigskip
\noindent
\textbf{Test 2 : Moderate ( depth 2 )}

\medskip

\noindent
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \implies (q \implies r)}{
  \ulcorner p \urcorner^{[1]}
&
\infer[\Rightarrow\textrm{-intro}^{[2]}]{q \implies r}{
  \ulcorner q \urcorner^{[2]}
&
\infer[axiom]{r}{}
}
}
$

\bigskip
\noindent
\textbf{Test 3 : Deep ( depth 3 )}

\medskip

\noindent
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \implies (q \implies (r \implies s))}{
  \ulcorner p \urcorner^{[1]}
&
\infer[\Rightarrow\textrm{-intro}^{[2]}]{q \implies (r \implies s)}{
  \ulcorner q \urcorner^{[2]}
&
\infer[\Rightarrow\textrm{-intro}^{[3]}]{r \implies s}{
  \ulcorner r \urcorner^{[3]}
&
\infer[axiom]{s}{}
}
}
}
$

\bigskip
\noindent
\textbf{Test 4 : Very Deep ( depth 4 )}

\medskip

\noindent
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \implies (q \implies (r \implies (s \implies t)))}{
  \ulcorner p \urcorner^{[1]}
&
\infer[\Rightarrow\textrm{-intro}^{[2]}]{q \implies (r \implies (s \implies t))}{
  \ulcorner q \urcorner^{[2]}
&
\infer[\Rightarrow\textrm{-intro}^{[3]}]{r \implies (s \implies t)}{
  \ulcorner r \urcorner^{[3]}
&
\infer[\Rightarrow\textrm{-intro}^{[4]}]{s \implies t}{
  \ulcorner s \urcorner^{[4]}
&
\infer[axiom]{t}{}
}
}
}
}
$

\bigskip
\noindent
\textbf{Test 5 : Horizontal siblings ( depth 2 )}

\medskip

\noindent
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{(p \land q) \implies r}{
  \ulcorner p \land q \urcorner^{[1]}
&
\infer[\land \mathrm{elim}]{r}{
  \ulcorner p \urcorner^{[1]}
&
\ulcorner q \urcorner^{[1]}
}
}
$
