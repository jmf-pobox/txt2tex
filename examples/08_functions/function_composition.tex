\subsection*{Example 1 : Basic Forward Composition}
\addcontentsline{toc}{subsection}{Example 1 : Basic Forward Composition}

\noindent Forward composition applies f first, then g:

\bigskip

\begin{axdef}
f : \nat \fun \nat \\
g : \nat \fun \nat \\
h : \nat \fun \nat
\where
\forall n : \nat @ f(n) = n + 1 \\
\forall n : \nat @ g(n) = n * 2 \\
h = f \comp g
\end{axdef}

\noindent h(n) = g(f(n)) = g(n + 1) = (n + 1) * $2 = 2$n + 2

\bigskip

\subsection*{Example 2 : Backward Composition}
\addcontentsline{toc}{subsection}{Example 2 : Backward Composition}

\noindent Backward composition applies the second function first (using comp):

\bigskip

\begin{axdef}
f2 : \nat \fun \nat \\
g2 : \nat \fun \nat \\
h2 : \nat \fun \nat
\where
\forall n : \nat @ f2(n) = n + 1 \\
\forall n : \nat @ g2(n) = n * 2 \\
h2 = f2 \comp g2
\end{axdef}

\noindent h2(n) = f2(g2(n)) = f2(n * 2) = (n * 2) + $1 = 2$n + 1

\bigskip

\subsection*{Example 3 : Comparison - Forward vs Backward}
\addcontentsline{toc}{subsection}{Example 3 : Comparison - Forward vs Backward}

\noindent Forward composition f $\semi$ g: Input $\rightarrow$ f $\rightarrow$ g $\rightarrow$ Output (f $\semi$ g)(x) = g(f(x))

\bigskip

\noindent Backward composition f o g: Input $\rightarrow$ g $\rightarrow$ f $\rightarrow$ Output (f o g)(x) = f(g(x))

\bigskip

\noindent Notice the reversal: $\semi$ applies left-to-right, o applies right-to-left.

\bigskip

\subsection*{Example 4 : Three - Function Composition}
\addcontentsline{toc}{subsection}{Example 4 : Three - Function Composition}

\begin{axdef}
addOne : \nat \fun \nat \\
double : \nat \fun \nat \\
square : \nat \fun \nat \\
pipeline : \nat \fun \nat
\where
\forall n : \nat @ addOne(n) = n + 1 \\
\forall n : \nat @ double(n) = 2 * n \\
\forall n : \nat @ square(n) = n * n \\
pipeline = addOne \comp double \comp square
\end{axdef}

\noindent pipeline(n) = square(double(addOne(n))) = square(double(n + 1)) = square(2(n + 1)) = (2(n + 1))\textasciicircum$\{\}$2

\bigskip

\subsection*{Example 5 : Identity Composition Laws}
\addcontentsline{toc}{subsection}{Example 5 : Identity Composition Laws}

\begin{axdef}
identity : \nat \fun \nat \\
anyFunc : \nat \fun \nat \\
leftId : \nat \fun \nat \\
rightId : \nat \fun \nat
\where
\forall n : \nat @ identity(n) = n \\
\forall n : \nat @ anyFunc(n) = n * n \\
leftId = identity \comp anyFunc \\
rightId = anyFunc \comp identity
\end{axdef}

\noindent $leftId = anyFunc$ (identity is left identity for $\semi$) $rightId = anyFunc$ (identity is right identity for $\semi$)

\bigskip

\subsection*{Example 6 : Associativity}
\addcontentsline{toc}{subsection}{Example 6 : Associativity}

\noindent Composition is associative:

\bigskip

\noindent (f $\semi$ g) $\semi$ $h = f$ $\semi$ (g $\semi$ h)

\bigskip

\noindent You can group compositions in any order—the result is the same.

\bigskip

\begin{axdef}
f3 : \nat \fun \nat \\
g3 : \nat \fun \nat \\
h3 : \nat \fun \nat \\
comp1 : \nat \fun \nat \\
comp2 : \nat \fun \nat
\where
\forall n : \nat @ f3(n) = n + 1 \\
\forall n : \nat @ g3(n) = n * 2 \\
\forall n : \nat @ h3(n) = n * n \\
comp1 = (f3 \comp g3) \comp h3 \\
comp2 = f3 \comp (g3 \comp h3)
\end{axdef}

\noindent $comp1 = comp2$ for all inputs.

\bigskip

\subsection*{Example 7 : Composition with Partial Functions}
\addcontentsline{toc}{subsection}{Example 7 : Composition with Partial Functions}

\begin{axdef}
safeSqrt : \nat \pfun \nat \\
addTen : \nat \fun \nat \\
composed : \nat \pfun \nat
\where
\forall n : \nat @ n * n = n \implies safeSqrt(n) = n \\
\forall n : \nat @ addTen(n) = n + 10 \\
composed = safeSqrt \comp addTen
\end{axdef}

\noindent composed is only defined where safeSqrt is defined, then adds 10 to the result.

\bigskip

\subsection*{Example 8 : Relation Composition}
\addcontentsline{toc}{subsection}{Example 8 : Relation Composition}

\noindent Composition works for relations too:

\bigskip

\begin{axdef}
parent : Person \rel Person \\
grandparent : Person \rel Person
\where
grandparent = parent \comp parent
\end{axdef}

\noindent grandparent relates a person to their grandparents via two parent links.

\subsection*{Example 12 : Partial Composition}
\addcontentsline{toc}{subsection}{Example 12 : Partial Composition}

\noindent When composing f : $A \fun B$ and g : $C \fun D$, we need ran f subseteq dom g.

\bigskip

\begin{axdef}
f4 : \nat \pfun \nat \\
g4 : \nat \pfun \nat \\
h4 : \nat \pfun \nat
\where
f4 = \{1 \mapsto 10, 2 \mapsto 20\} \\
g4 = \{10 \mapsto 100, 20 \mapsto 200\} \\
h4 = f4 \comp g4
\end{axdef}

\noindent h4 = $\{1 \mapsto 100, 2 \mapsto 200\}$. Composition is defined where ranges align.

\bigskip

\subsection*{Example 13 : Monoid Structure}
\addcontentsline{toc}{subsection}{Example 13 : Monoid Structure}

\noindent Functions from $A \fun A$ with composition form a monoid: 1. Composition is associative: (f $\semi$ g) $\semi$ $h = f$ $\semi$ (g $\semi$ h) 2. Identity element: id(x) = x, id $\semi$ $f = f = f$ $\semi$ id 3. Closure: composing $A \fun A$ functions gives another $A \fun A$ function

\bigskip

\subsection*{Example 14 : Practical Example - Data Processing Pipeline}
\addcontentsline{toc}{subsection}{Example 14 : Practical Example - Data Processing Pipeline}

\noindent In practice, you might compose functions to create data processing pipelines: validate $\semi$ process $\semi$ format. Each stage transforms data through a series of operations. The notation f $\semi$ g $\semi$ h means "apply f, then g, then h" which creates a pipeline from input to output.

\subsection*{Example 15 : Composition Operator Precedence}
\addcontentsline{toc}{subsection}{Example 15 : Composition Operator Precedence}

\noindent Composition operators bind tighter than logical operators:

\bigskip

\noindent f $\semi$ $g = h$ means (f $\semi$ g) = h, not f $\semi$ ($g = h$)

\bigskip

\noindent Use parentheses to clarify: (f $\semi$ g) or f $\semi$ (g $\semi$ h)

\bigskip

\subsection*{Example 16 : Self - Composition}
\addcontentsline{toc}{subsection}{Example 16 : Self - Composition}

\begin{axdef}
doubleIt : \nat \fun \nat \\
quadrupleIt : \nat \fun \nat
\where
\forall n : \nat @ doubleIt(n) = 2 * n \\
quadrupleIt = doubleIt \comp doubleIt
\end{axdef}

\noindent quadrupleIt(n) = doubleIt(doubleIt(n)) = 2 * (2 * n) = 4n

\subsection*{Example 18 : Functional Programming Style}
\addcontentsline{toc}{subsection}{Example 18 : Functional Programming Style}

\noindent Composition enables a functional programming style:

\bigskip

\noindent Instead of: $result = h$(g(f(x))) Use: $pipeline = f$ $\semi$ g $\semi$ h; $result = pipeline$(x)

\bigskip

\noindent This separates pipeline definition from application, improving modularity.

\bigskip

\subsection*{Example 19 : Best Practices}
\addcontentsline{toc}{subsection}{Example 19 : Best Practices}

\noindent When using composition: 1. Choose forward ($\semi$) or backward (o) consistently in your codebase 2. Document the data flow direction 3. Break complex pipelines into named intermediate steps 4. Type-check that domains and ranges align 5. Use composition to create reusable pipelines 6. Test individual functions before composing them 7. Remember: $\semi$ reads left-to-right (natural for pipelines), o reads right-to-left (traditional math notation)
