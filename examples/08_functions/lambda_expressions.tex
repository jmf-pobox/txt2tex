\section*{Phase 11d: Lambda Expressions}
\addcontentsline{toc}{section}{Phase 11d: Lambda Expressions}

\noindent Lambda expressions provide a concise way to define anonymous functions.

\bigskip

\medskip

\subsubsection*{(a)}
\addcontentsline{toc}{subsubsection}{(a)}

\noindent Simple lambda expression

\bigskip

\noindent
$(\lambda x : \nat @ x \bsup 2 \esup)$


\medskip

\subsubsection*{(b)}
\addcontentsline{toc}{subsubsection}{(b)}

\noindent Multi-variable lambda

\bigskip

\noindent
$(\lambda x, y : \nat @ x \land y)$


\medskip

\subsubsection*{(c)}
\addcontentsline{toc}{subsubsection}{(c)}

\noindent Lambda with comparison

\bigskip

\noindent
$(\lambda x : \nat @ x > 0)$


\medskip

\subsubsection*{(d)}
\addcontentsline{toc}{subsubsection}{(d)}

\noindent Nested lambda expression

\bigskip

\noindent
$(\lambda x : X @ (\lambda y : Y @ x \land y))$


\medskip

\subsubsection*{(e)}
\addcontentsline{toc}{subsubsection}{(e)}

\noindent Lambda with quantifier in body

\bigskip

\noindent
$(\lambda x : X @ \forall y : Y @ x \lor y)$
