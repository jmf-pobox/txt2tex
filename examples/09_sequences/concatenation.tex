\section*{Sequence Concatenation}
\addcontentsline{toc}{section}{Sequence Concatenation}

\subsection*{Example 1 : Simple Concatenation ( ASCII )}
\addcontentsline{toc}{subsection}{Example 1 : Simple Concatenation ( ASCII )}

\noindent The caret operator after a sequence means concatenation:

\bigskip

\noindent
$\langle a \rangle \cat \langle b \rangle$


\noindent
$\langle 1, 2 \rangle \cat \langle 3, 4 \rangle$


\noindent
$\langle x \rangle \cat \langle \rangle$


\subsection*{Example 2 : Concatenation with Variables}
\addcontentsline{toc}{subsection}{Example 2 : Concatenation with Variables}

\noindent Concatenation with sequence variables:

\bigskip

\noindent
$\langle x \rangle \cat s$


\noindent
$s \cat t$


\noindent
$\langle \rangle \cat t$


\subsection*{Example 3 : Cons Pattern}
\addcontentsline{toc}{subsection}{Example 3 : Cons Pattern}

\noindent The cons pattern builds sequences incrementally:

\bigskip

\noindent
$\langle x \rangle \cat \langle \rangle$


\noindent
$\langle x \rangle \cat \langle y \rangle \cat \langle z \rangle$


\noindent
$\langle 1 \rangle \cat \langle 2 \rangle \cat \langle 3 \rangle \cat \langle \rangle$


\subsection*{Example 4 : Nested Concatenation}
\addcontentsline{toc}{subsection}{Example 4 : Nested Concatenation}

\noindent Concatenation is associative:

\bigskip

\noindent
$(\langle a \rangle \cat \langle b \rangle) \cat \langle c \rangle$


\noindent
$\langle a \rangle \cat (\langle b \rangle \cat \langle c \rangle)$


\noindent
$(s \cat t) \cat u$
