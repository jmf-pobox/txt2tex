\section*{Atomic Binding}

\noindent
$\lblot~name == s.name~\rblot$


\section*{Binding in Set Comprehension ( Q2 ( a ) style )}

\noindent
$\{~ s : Ship | s.launched < 1921 @ \lblot~name == s.name~\rblot ~\}$


\section*{Multi - Component Binding}

\noindent
$\lblot~name == s.name, displacement == c.displacement, numGuns == c.numGuns~\rblot$


\section*{Multi - Variable Comprehension with Binding ( Q2 ( d ) style )}

\noindent
$\{~ s : Ship ; c : Class | s.class = c.class @ \lblot~name == s.name, displacement == c.displacement, numGuns == c.numGuns~\rblot ~\}$


\section*{Empty Binding}

\noindent
$\lblot~\rblot$
