// Primary Keys (Phase 2.1)
// The pk prefix marks the primary-key attribute in a named schema body.
// txt2tex emits a PK statement after the schema box — fuzz-compatible.

TITLE: Primary Keys in Relational Schemas
AUTHOR: txt2tex example

given ClassName, ShipName, BattleName, CountryName, PersonID, CentreID

=== Single Primary Key ===

// The pk prefix marks the attribute; a PK line appears after the schema box.

schema Class
  pk class : ClassName
  country : CountryName
  numGuns : N
  bore : N
  displacement : N
end

schema Ship
  pk name : ShipName
  class : ClassName
  launched : N
end

schema Battle
  pk name : BattleName
  date : N
end

=== Composite Primary Key ===

// Two pk lines list both attributes in the PK set.

schema CentreStaff
  pk centreId : CentreID
  pk staffId : PersonID
end
