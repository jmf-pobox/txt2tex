\section*{Single Primary Key}

\begin{schema}{Book}
bookId : BookId \\
isbn : ISBN \\
pages : \nat \\
year : \nat
\end{schema}
\noindent$\mathrm{PK}(\mathrm{Book}) = \{bookId\}$

\smallskip

\begin{schema}{Member}
memberId : MemberId \\
age : \nat
\end{schema}
\noindent$\mathrm{PK}(\mathrm{Member}) = \{memberId\}$

\smallskip

\begin{schema}{Loan}
loanId : LoanId \\
bookId : BookId \\
memberId : MemberId \\
daysOut : \nat
\end{schema}
\noindent$\mathrm{PK}(\mathrm{Loan}) = \{loanId\}$

\smallskip

\section*{Composite Primary Key}

\begin{schema}{LoanHistory}
bookId : BookId \\
memberId : MemberId
\end{schema}
\noindent$\mathrm{PK}(\mathrm{LoanHistory}) = \{bookId, memberId\}$
