\section*{Predicate Logic Phase 6 Tests}

\bigskip
\noindent
\textbf{Test 1 : Simple Quantifiers}

\medskip

(a)
\par
\vspace{11pt}
$\forall x \colon N \bullet x > 0$

\medskip

(b)
\par
\vspace{11pt}
$\exists y \colon N \bullet y = 0$

\medskip

(c)
\par
\vspace{11pt}
$\exists_1 z \colon N \bullet z < 10$

\medskip

\bigskip
\noindent
\textbf{Test 2 : Multi Variable Quantifiers}

\medskip

(a)
\par
\vspace{11pt}
$\forall x, y \colon N \bullet x > y$

\medskip

(b)
\par
\vspace{11pt}
$\exists a, b \colon N \bullet a > 0 \land b > 0$

\medskip

(c)
\par
\vspace{11pt}
$\forall m, n, p \colon N \bullet m \geq n \land n \geq p$

\medskip

\bigskip
\noindent
\textbf{Test 3 : Nested Quantifiers}

\medskip

(a)
\par
\vspace{11pt}
$\forall x \colon N \bullet \exists y \colon N \bullet x > y$

\medskip

(b)
\par
\vspace{11pt}
$\exists x \colon N \bullet \forall y \colon N \bullet x \leq y$

\medskip
