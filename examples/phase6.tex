\section*{Predicate Logic Phase 6 Tests}

\bigskip
\noindent
\textbf{Test 1 : Simple Quantifiers}

\medskip

\noindent
\hangindent=2em
(a) $\forall x \colon \mathbb{N} \bullet x > 0$

\medskip
\noindent
\hangindent=2em
(b) $\exists y \colon \mathbb{N} \bullet y = 0$

\medskip
\noindent
\hangindent=2em
(c) $\exists_1 z \colon \mathbb{N} \bullet z < 10$

\medskip
\bigskip
\noindent
\textbf{Test 2 : Multi Variable Quantifiers}

\medskip

\noindent
\hangindent=2em
(a) $\forall x, y \colon \mathbb{N} \bullet x > y$

\medskip
\noindent
\hangindent=2em
(b) $\exists a, b \colon \mathbb{N} \bullet a > 0 \land b > 0$

\medskip
\noindent
\hangindent=2em
(c) $\forall m, n, p \colon \mathbb{N} \bullet m \geq n \land n \geq p$

\medskip
\bigskip
\noindent
\textbf{Test 3 : Nested Quantifiers}

\medskip

\noindent
\hangindent=2em
(a) $\forall x \colon \mathbb{N} \bullet \exists y \colon \mathbb{N} \bullet x > y$

\medskip
\noindent
\hangindent=2em
(b) $\exists x \colon \mathbb{N} \bullet \forall y \colon \mathbb{N} \bullet x \leq y$
