}
\end{center}
\bigskip

\subsubsection*{(b)}
\addcontentsline{toc}{subsubsection}{(b)}

\noindent
$Multiple(equivalence)(steps)$


\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\begin{array}{l@{\hspace{2em}}r}
(p \land q) \lor (p \land r) \\
\Leftrightarrow p \land (q \lor r) & [\mbox{distributive}] \\
\Leftrightarrow (q \lor r) \land p & [\mbox{commutative}] \\
\Leftrightarrow p \land (r \lor q) & [\mbox{commutative}]
\end{array}$%
}
\end{center}

$Mixed(justifications)$


\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\begin{array}{l@{\hspace{2em}}r}
p \Leftrightarrow q \\
\Leftrightarrow (p \implies q) \land (q \implies p) & [\mbox{definition}] \\

\end{array}$%
}
