\section*{Equivalence Chains}

\section*{Solution 1}
\addcontentsline{toc}{section}{Solution 1}

\subsection*{(a)}

\noindent
$Commutative(property)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      p \land q \\
      \Leftrightarrow q \land p & [\mbox{commutative}]
    \end{array}$%
  }
\end{center}
\bigskip

\medskip

\subsection*{(b)}

\noindent
$De(Morgan)(law)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      \lnot (p \land q) \\
      \Leftrightarrow \lnot p \lor \lnot q & [\mbox{De Morgan}] \\
      \Leftrightarrow \lnot p \lor \lnot q & [\mbox{parentheses}]
    \end{array}$%
  }
\end{center}
\bigskip

\section*{Solution 2}
\addcontentsline{toc}{section}{Solution 2}

\subsection*{(a)}

\noindent
$Definition$

\noindent
$of(implication)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      p \implies q \\
      \Leftrightarrow \lnot p \lor q & [\mbox{definition}]
    \end{array}$%
  }
\end{center}
\bigskip

\medskip

\subsection*{(b)}

\noindent
$Multiple(equivalence)(steps)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      (p \land q) \lor (p \land r) \\
      \Leftrightarrow p \land (q \lor r) & [\mbox{distributive}] \\
      \Leftrightarrow (q \lor r) \land p & [\mbox{commutative}] \\
      \Leftrightarrow p \land (r \lor q) & [\mbox{commutative}]
    \end{array}$%
  }
\end{center}
\bigskip

\section*{Solution 3}
\addcontentsline{toc}{section}{Solution 3}

\subsection*{(a)}

\noindent
$Equivalence(chain)(without)(justifications)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      p \lor p \\
      \Leftrightarrow p
    \end{array}$%
  }
\end{center}
\bigskip

\medskip

\subsection*{(b)}

\noindent
$Mixed(justifications)$

\begin{center}
  \adjustbox{max width=\textwidth}{%
    $\displaystyle
    \begin{array}{l@{\hspace{2em}}r}
      p \Leftrightarrow q \\
      \Leftrightarrow (p \implies q) \land (q \implies p) &
      [\mbox{definition}] \\
      \Leftrightarrow (\lnot p \lor q) \land (\lnot q \lor p)
    \end{array}$%
  }
\end{center}
\bigskip
