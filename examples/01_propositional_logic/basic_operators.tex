\section*{Propositional Logic}

\bigskip
\noindent
\textbf{Solution 1}

\medskip

\noindent
\hangindent=2em
(a) $p \land q \implies p$

\medskip
(b)
\par
\vspace{11pt}
\begin{tabular}{c|c|c}
$p$ & $q$ & $\mathbf{p \Rightarrow q}$ \\
\hline
\textit{t} & \textit{t} & \textbf{t} \\
\textit{t} & \textit{f} & \textbf{f} \\
\textit{f} & \textit{t} & \textbf{t} \\
\textit{f} & \textit{f} & \textbf{t} \\
\end{tabular}

\medskip

\noindent
\hangindent=2em
(c) $\lnot (p \land q) \iff \lnot p \lor \lnot q$

\noindent
\textbf{Solution 2}

\medskip

\noindent
\hangindent=2em
(a) $(p \lor q) \land r \iff (p \land r) \lor (q \land r)$

\medskip
(b)
\par
\vspace{11pt}
\begin{tabular}{c|c|c}
$p$ & $q$ & $\mathbf{p \Leftrightarrow q}$ \\
\hline
\textit{t} & \textit{t} & \textbf{t} \\
\textit{t} & \textit{f} & \textbf{f} \\
\textit{f} & \textit{t} & \textbf{f} \\
\textit{f} & \textit{f} & \textbf{t} \\
\end{tabular}

\medskip

\noindent
\hangindent=2em
(c) $p \implies (q \implies r)$
