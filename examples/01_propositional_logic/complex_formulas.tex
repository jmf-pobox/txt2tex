\section*{Phase 3 : Quantifiers and Mathematical Notation}

\bigskip
\noindent
\textbf{Example 1 : Quantifiers}

\medskip

\noindent
\hangindent=2em
(a) $\forall x : \nat @ x \geq 0$

\medskip
\noindent
\hangindent=2em
(b) $\exists y : \nat @ y = 0$

\medskip
\noindent
\hangindent=2em
(c) $\forall n : \nat @ n \geq 0 \land n < 100$

\medskip
\bigskip
\noindent
\textbf{Example 2 : Subscripts and Superscripts}

\medskip

\noindent
\hangindent=2em
(a) $x^2$

\medskip
\noindent
\hangindent=2em
(b) $a_1$

\medskip
\noindent
\hangindent=2em
(c) $x_i^2$

\medskip
\bigskip
\noindent
\textbf{Example 3 : Comparison Operators}

\medskip

\noindent
\hangindent=2em
(a) $x < 10$

\medskip
\noindent
\hangindent=2em
(b) $x \geq 0 \land x \leq 100$

\medskip
\noindent
\hangindent=2em
(c) $y > 5 \lor y = 0$

\medskip
\bigskip
\noindent
\textbf{Example 4 : Set Operators}

\medskip

\noindent
\hangindent=2em
(a) $x \in \nat$

\medskip
\noindent
\hangindent=2em
(b) $A \subseteq B$

\medskip
\noindent
\hangindent=2em
(c) $A \cup B \cap C$

\medskip
\bigskip
\noindent
\textbf{Example 5 : Complex Expressions}

\medskip

(a)
\par
\vspace{11pt}
\[
\begin{array}{ll@{\hspace{2em}}l}
& \forall x : \nat @ x \in \nat \land x \geq 0 \\
&\Leftrightarrow \forall x : \nat @ x \geq 0
\end{array}
\]

\medskip

\noindent
\hangindent=2em
(b) $\forall i : \nat @ a_i > 0 \implies a_i^2 > 0$

\medskip
(c)
\par
\vspace{11pt}
\begin{tabular}{c|c|c}
$p$ & $q$ & $\mathbf{p \Rightarrow q}$ \\
\hline
\textit{t} & \textit{t} & \textbf{t} \\
\textit{t} & \textit{f} & \textbf{f} \\
\textit{f} & \textit{t} & \textbf{t} \\
\textit{f} & \textit{f} & \textbf{t} \\
\end{tabular}
