\noindent We prove that conjunction implies each of its components.

\bigskip

\subsection*{Part ( a ) : p land q implies p}
\addcontentsline{toc}{subsection}{Part ( a ) : p land q implies p}

\begin{center}
\adjustbox{max width=\textwidth}{%
$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \land q \implies p}{
  \infer[\land\textrm{-elim-1}]{p}{\ulcorner p \land q \urcorner^{[1]}}
}
$%
}
\end{center}
\bigskip

\subsection*{Part ( b ) : p land q implies q}
\addcontentsline{toc}{subsection}{Part ( b ) : p land q implies q}

\begin{center}

$\displaystyle
\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \land q \implies q}{
  \infer[\land\textrm{-elim-2}]{q}{\ulcorner p \land q \urcorner^{[1]}}
}
$%
}
\end{center}
