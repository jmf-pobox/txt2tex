\section*{Function Application Examples}

\begin{zed}[Item, Dog, Person, X]\end{zed}

\bigskip

This file demonstrates function $application \in Z$ notation definitions.

\bigskip

\bigskip
\noindent
\textbf{Example 1 : Basic function definitions}

\medskip

\begin{axdef}
square : \nat \fun \nat \\
double : \nat \fun \nat
\where
\forall n : \nat @ square(n) = n * n \\
\forall n : \nat @ double(n) = 2 * n
\end{axdef}

\bigskip
\noindent
\textbf{Example 2 : Functions with multiple arguments}

\medskip

\begin{axdef}
add : \nat \cross \nat \fun \nat \\
maximum : \nat \cross \nat \fun \nat
\where
\forall a, b : \nat @ add(a, b) = a + b \\
\forall a, b : \nat @ (maximum(a, b) = a \lor maximum(a, b) = b) \land maximum(a, b) \geq a \land maximum(a, b) \geq b
\end{axdef}

\bigskip
\noindent
\textbf{Example 3 : Generic functions}

\medskip

\begin{gendef}[X]
  collection: \seq X
\end{gendef}

\bigskip
\noindent
\textbf{Example 4 : Function application in quantifiers}

\medskip

\begin{axdef}
posFunc : \nat \fun \nat
\where
\forall x : \nat @ posFunc(x) > 0 \implies posFunc(x) > 0
\end{axdef}

\bigskip
\noindent
\textbf{Example 5 : Schema with function type}

\medskip

\begin{schema}{Heights}
height : Person \fun \nat
\where
\forall p : Person @ height(p) > 0 \land height(p) < 300
\end{schema}

\bigskip
\noindent
\textbf{Example 6 : Nested function application}

\medskip

\begin{axdef}
func1 : \nat \fun \nat \\
func2 : \nat \fun \nat \\
func3 : \nat \fun \nat
\where
\forall x : \nat @ func1(func2(func3(x))) \geq 0
\end{axdef}
