\section*{Phase 11 a : Function Types}

\begin{zed}[X, Y, Z]
\end{zed}

\noindent Total functions (every element in domain has a mapping):

\bigskip

\begin{axdef}
  totalF1 : X \fun Y \\
  totalF2 : \nat \fun \nat
\end{axdef}

\noindent Partial functions (some elements may not have a mapping):

\bigskip

\begin{axdef}
  partialF : X \pfun Y
\end{axdef}

\noindent Injections (one-to-one mappings):

\bigskip

\begin{axdef}
  injTotal : X \inj Y \\
  injPartial : X \pinj Y
\end{axdef}

\noindent Surjections (onto mappings):

\bigskip

\begin{axdef}
  surjTotal : X \surj Y \\
  surjPartial : X \psurj Y
\end{axdef}

\noindent Bijections $(one - to - one \land onto)$:

\bigskip

\begin{axdef}
  bijection : X \bij Y
\end{axdef}

\noindent Mixed operators $(relations \land function(types)(work)(together))$:

\bigskip

\begin{axdef}
  relation : X \rel Y \\
  function : X \fun Y
\end{axdef}
