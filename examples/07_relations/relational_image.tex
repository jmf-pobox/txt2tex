\subsection*{Phase 11 . 8 : Relational Image}
\addcontentsline{toc}{subsection}{Phase 11 . 8 : Relational Image}

\noindent The relational image $(R \limg S \rimg)$ gives the image of set S under relation R.

\bigskip

\subsection*{Example 1 : Simple relational image}
\addcontentsline{toc}{subsection}{Example 1 : Simple relational image}

\noindent
$(R \limg S \rimg)$


\subsection*{Example 2 : Relational image with set literal}
\addcontentsline{toc}{subsection}{Example 2 : Relational image with set literal}

\noindent
$(R \limg \{1, 2, 3\} \rimg)$


\subsection*{Example 3 : Relational image with singleton set}
\addcontentsline{toc}{subsection}{Example 3 : Relational image with singleton set}

\noindent
$(parentOf \limg \{john\} \rimg)$


\subsection*{Example 4 : Relational image with composition}
\addcontentsline{toc}{subsection}{Example 4 : Relational image with composition}

\noindent
$((R \semi S) \limg A \rimg)$


\subsection*{Example 5 : Relational image with inverse}
\addcontentsline{toc}{subsection}{Example 5 : Relational image with inverse}

\noindent
$(R \inv \limg S \rimg)$
