\section*{Phase 7 : Equality and Special Operators}

\bigskip
\noindent
\textbf{Test 1 : Mu Operator}

\medskip

(a)
\par
\vspace{11pt}
$\mu x \colon N \bullet x^2 = 4 \land x > 0$

\medskip

(b)
\par
\vspace{11pt}
$\mu x \bullet x > 0 \land x < 10$

\medskip

\bigskip
\noindent
\textbf{Test 2 : Not Equal Operator}

\medskip

(a)
\par
\vspace{11pt}
$x \neq 0$

\medskip

(b)
\par
\vspace{11pt}
$x \neq 0 \land y \neq 0$

\medskip

(c)
\par
\vspace{11pt}
$x \neq y \lor y \neq z$

\medskip

\bigskip
\noindent
\textbf{Test 3 : Not In Operator}

\medskip

(a)
\par
\vspace{11pt}
$x \notin S$

\medskip

(b)
\par
\vspace{11pt}
$x \in A \land x \notin B$

\medskip

(c)
\par
\vspace{11pt}
$x \notin S \Leftrightarrow \lnot (x \in S)$

\medskip

\bigskip
\noindent
\textbf{Test 4 : One Point Rule}

\medskip

\vspace{-10pt}
\begin{align*}
\exists y \colon N \bullet y = 0 \land x \neq y \\
&\Leftrightarrow 0 \in N \land x \neq 0 \\
&\Leftrightarrow x \neq 0
\end{align*}

\bigskip
\noindent
\textbf{Test 5 : Complex Reasoning}

\medskip

(a)
\par
\vspace{11pt}
$\mu x \colon N \bullet x^2 = 9 \land x \neq 3$

\medskip

(b)
\par
\vspace{11pt}
$\forall x \colon N \bullet \exists y \colon N \bullet x \neq y$

\medskip

(c)
\par
\vspace{11pt}
$\forall x \colon N \bullet x \in S \Rightarrow x \notin T$

\medskip

\bigskip
\noindent
\textbf{Test 6 : Mixed Operators}

\medskip

(a)
\par
\vspace{11pt}
$x = y \lor x \neq y$

\medskip

(b)
\par
\vspace{11pt}
$x \in A \lor x \notin A$

\medskip

(c)
\par
\vspace{11pt}
$\mu x \colon N \bullet x \in S \land x \neq 0 \land x^2 < 10$

\medskip
