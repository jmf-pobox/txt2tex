\textbf{Question 2}

\medskip

(a)
\par
\vspace{11pt}
\vspace{-10pt}
\begin{align*}
p \Rightarrow \lnot p \Rightarrow p \\
&\Leftrightarrow p \Rightarrow \lnot \lnot p \lor p && \text{[$\Rightarrow$ disjunction]} \\
&\Leftrightarrow p \Rightarrow p \lor p && \text{[$\lnot$ $\lnot$]} \\
&\Leftrightarrow p \Rightarrow p && \text{[idempotence of p]} \\
&\Leftrightarrow true && \text{[tautology]}
\end{align*}

\medskip

(b)
\par
\vspace{11pt}
\vspace{-10pt}
\begin{align*}
\lnot p \Rightarrow \lnot q \\
&\Leftrightarrow \lnot \lnot p \lor \lnot q && \text{[$\Rightarrow$ disjunction]} \\
&\Leftrightarrow p \lor \lnot q && \text{[$\lnot$ $\lnot$]} \\
&\Leftrightarrow \lnot q \lor p && \text{[commutativity of $\lor$]} \\
&\Leftrightarrow q \Rightarrow p && \text{[$\Rightarrow$ disjunction]}
\end{align*}

\medskip

(c)
\par
\vspace{11pt}
\vspace{-10pt}
\begin{align*}
(p \Rightarrow r) \land (q \Rightarrow r) \\
&\Leftrightarrow (\lnot p \lor r) \land (\lnot q \lor r) && \text{[$\Rightarrow$ disjunction]} \\
&\Leftrightarrow r \land (\lnot p \lor \lnot q) && \text{[distributivity 2]} \\
&\Leftrightarrow (\lnot p \lor \lnot q) \land r && \text{[associativity of $\lor$]} \\
&\Leftrightarrow p \lor q \Rightarrow r
\end{align*}
