\section*{Propositional Logic}

\bigskip
\noindent
\textbf{Solution 3}

(a)
\begin{align*}
p \\
&\Leftrightarrow p \lor \lnot p && \text{[$\lor$ introduction]} \\
&\Leftrightarrow \lnot p \Rightarrow p && \text{[$\Rightarrow$ disjunction]}
\end{align*}

\medskip

(c)
\begin{align*}
(p \Rightarrow r) \land (q \Rightarrow r) \\
&\Leftrightarrow (\lnot p \lor r) \land (\lnot q \lor r) && \text{[$\Rightarrow$ disjunction]} \\
&\Leftrightarrow r \land (\lnot p \lor \lnot q) && \text{[distributivity 2]} \\
&\Leftrightarrow (\lnot p \lor \lnot q) \land r && \text{[associativity of $\lor$]} \\
&\Leftrightarrow p \lor q \Rightarrow r
\end{align*}
