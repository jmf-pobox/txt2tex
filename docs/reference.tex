% ── PROPOSITIONAL LOGIC ─────────────────────────
\section{Propositional Logic}

\begin{tabularx}{\columnwidth}{LR}
p land q             & $p \land q$ \\
p lor q              & $p \lor q$ \\
lnot p               & $\lnot p$ \\
p => q               & $p \Rightarrow q$ \\
p <=> q              & $p \Leftrightarrow q$ \\
\end{tabularx}

{\footnotesize\textbf{Note:} Use \kw{land}, \kw{lor}, \kw{lnot}. English \texttt{and}/\texttt{or}/\texttt{not} are not supported.}

\smallskip
\needspace{14\baselineskip}
\textbf{Example — truth table (validate De~Morgan).}

\begin{exwrite}%
\begin{verbatim}
TRUTH TABLE:
p | q | p land q | lnot (p land q) | lnot p | lnot q | lnot p lor lnot q
T | T | T | F | F | F | F
T | F | F | T | F | T | T
F | T | F | T | T | F | T
F | F | F | T | T | T | T
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
\begin{tabular}{ccccccc}
$p$ & $q$ & $p \land q$ & $\lnot (p \land q)$ & $\lnot p$ & $\lnot q$ & $\lnot p \lor \lnot q$ \\ \hline
T & T & T & F & F & F & F \\
T & F & F & T & F & T & T \\
F & T & F & T & T & F & T \\
F & F & F & T & T & T & T \\
\end{tabular}
\end{center}
\end{exget}

{\scriptsize Build intermediate columns step by step.  Columns 4 and 7 match on every row, so $\lnot (p \land q) \Leftrightarrow \lnot p \lor \lnot q$ (De~Morgan).}

\needspace{14\baselineskip}
\textbf{Example — EQUIV chain.}

\begin{exwrite}%
\begin{verbatim}
EQUIV:
lnot (p land (q lor r))
lnot p lor lnot (q lor r)         [De Morgan]
lnot p lor (lnot q land lnot r)   [De Morgan]
(lnot p lor lnot q) land
    (lnot p lor lnot r)           [distribute]
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\begin{array}{l@{\hspace{2em}}r}
\lnot (p \land (q \lor r)) \\
\Leftrightarrow \lnot p \lor \lnot (q \lor r)
  & [\mbox{De Morgan}] \\
\Leftrightarrow \lnot p \lor (\lnot q \land \lnot r)
  & [\mbox{De Morgan}] \\
\Leftrightarrow (\lnot p \lor \lnot q) \land (\lnot p \lor \lnot r)
  & [\mbox{distribute}]
\end{array}$
\end{center}
\end{exget}

{\scriptsize \kw{EQUIV:} and its alias \kw{ARGUE:} join steps with $\Leftrightarrow$. Use for predicate equivalences.}

\needspace{14\baselineskip}
\textbf{Example — EQUAL chain.}

\begin{exwrite}%
\begin{verbatim}
EQUAL:
#(s ^ <a, b, c>)
#s + #<a, b, c>            [#-concat]
#s + 3                      [literal length]
3 + #s                      [+ commutative]
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\begin{array}{l@{\hspace{1.5em}}r}
\#(s \cat \langle a, b, c \rangle) \\
= \#s + \#\langle a, b, c \rangle & [\mbox{\#-concat}] \\
= \#s + 3 & [\mbox{literal length}] \\
= 3 + \#s & [\mbox{$+$ commutative}]
\end{array}$
\end{center}
\end{exget}

{\scriptsize \kw{EQUAL:} joins steps with $=$. Use when steps are arithmetic/set-valued, not propositional equivalences.}

\begin{tabularx}{\columnwidth}{LR}

\begin{exwrite}%
\begin{verbatim}

\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\forall (x, y) : \nat \cross \nat @ x + y \geq 0$
\end{center}
\end{exget}

% ── Z ───────────────────────────────────────────
\section{Z}

{\scriptsize A Z specification is a sequence of paragraphs.  Each paragraph either introduces names or constrains them.  The kinds below cover the building blocks; \kw{schema} (the most-used kind) gets its own section.}

\smallskip
\needspace{4\baselineskip}
\textbf{Given (carrier) type.} An opaque set whose elements are not enumerated --- used when identity matters but structure does not.

\begin{tabularx}{\columnwidth}{LR}
given A, B           & $[A, B]$ \\
\end{tabularx}

\smallskip
\needspace{4\baselineskip}
\textbf{Free type.} A named type with a fixed enumeration of constructors.  Branches can carry data via \texttt{<\ldots>}.

\begin{tabularx}{\columnwidth}{LR}
Status ::= active | inactive  & $Status ::= active \mid inactive$ \\
\end{tabularx}

\smallskip
\needspace{4\baselineskip}
\textbf{Abbreviation.} Binds a name to an expression for reuse --- no new type.

\begin{tabularx}{\columnwidth}{LR}
MaxSize == 100              & $MaxSize == 100$ \\
\end{tabularx}

\smallskip
\needspace{14\baselineskip}
\textbf{Axiomatic definition.} Introduces global constants constrained by predicates.  Form: \kw{axdef} declarations \kw{where} predicates \kw{end}.  (See \textit{Generic schemas} for the \kw{gendef [X]} variant.)

\begin{exwrite}%
\begin{verbatim}
axdef
  MaxSize : N
  Reset : N -> N
where
  MaxSize > 0
  Reset(MaxSize) = 0
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{axdef}
MaxSize : \nat \\
Reset : \nat \fun \nat
\where
MaxSize > 0 \\
Reset(MaxSize) = 0
\end{axdef}
\end{exget}

\smallskip
\needspace{4\baselineskip}
\textbf{Source-level containers.} The forms above stand alone (the engine wraps them in \kw{zed}).  Use explicit containers to bundle several paragraphs or align multiple free types.

\begin{tabularx}{\columnwidth}{LR}
zed \ldots end       & bundle several standalone paragraphs (unboxed) \\
syntax \ldots end    & column-aligned multi-type free definitions \\
\end{tabularx}

\smallskip
\needspace{14\baselineskip}
\textbf{Example — zed bundle.}  Combines a \textit{given type}, a \textit{free type}, and an \textit{abbreviation} in one unboxed block.

\begin{exwrite}%
\begin{verbatim}
zed
  given Entry
  Status ::= active | inactive
  MaxSize == 100
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{zed}
[Entry] \also
Status ::= active | inactive \also
MaxSize == 100
\end{zed}
\end{exget}

\smallskip
\needspace{14\baselineskip}
\textbf{Example — syntax (aligned free types).}  Two mutually-recursive free types defined together with their \texttt{::=} columns aligned.

\begin{exwrite}%
\begin{verbatim}
syntax
  OP ::= plus | minus | times
  Expr ::= const<N> | binop<OP cross Expr>
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{tabular}{rcl}
$OP$   & $::=$ & $plus \mid minus \mid times$ \\
$Expr$ & $::=$ & $const \ldata \nat \rdata \mid binop \ldata OP \cross Expr \rdata$
\end{tabular}
\end{exget}

{\scriptsize Blank lines between definitions inside \kw{syntax}/\kw{zed} emit \texttt{\textbackslash also} (visual group separator).}

\section{Schemas}

{\scriptsize A named schema introduces a signature (declarations) and an invariant (where clause).}

\begin{exwrite}%
\begin{verbatim}
schema Counter
  size : N
  nums : seq N
where
  size = # nums
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{schema}{Counter}
size : \nat \\
nums : \seq \nat
\where
size = \# nums
\end{schema}
\end{exget}

\smallskip
\needspace{14\baselineskip}
\textbf{Where-clause patterns.}

\needspace{14\baselineskip}
\textbf{1. Set comprehension.}
\begin{exwrite}%
\begin{verbatim}
schema Inventory
  items : seq N
  stock : N
where
  stock = # {x : ran items | x > 0}
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{schema}{Inventory}
items : \seq \nat \\
stock : \nat
\where
stock = \# \{~ x : \ran items \mid x > 0 ~\}
\end{schema}
\end{exget}

\needspace{14\baselineskip}
\textbf{2. Quantifier.}
\begin{exwrite}%
\begin{verbatim}
schema NonNeg
  s : seq Z
where
  forall i : 1..#s | s(i) >= 0
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{schema}{NonNeg}
s : \seq \num
\where
\forall i : 1 \upto \# s @ s(i) \geq 0
\end{schema}
\end{exget}

\smallskip
\needspace{14\baselineskip}
\textbf{Generic schemas.}

\begin{exwrite}%
\begin{verbatim}
gendef [X]
  empty : F X
where
  empty = {}
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{gendef}[X]
empty : \finset X
\where
empty = \{\}
\end{gendef}
\end{exget}

{\scriptsize $[X]$ is the type parameter; instantiate at use with \texttt{empty[T]}.}

\end{exwrite}

\end{zed}
\end{exget}

\smallskip
\needspace{16\baselineskip}
\textbf{Example — promotion schema (\kw{theta} lifts local to global).}

{\scriptsize \kw{Bank} has \texttt{counters : ID +-> Counter} (partial — only some IDs are bound); promotion lifts a local \kw{Bump} into one entry:}

\begin{exwrite}%
\begin{verbatim}
schema PromoteBump
  Delta Bank
  Delta Counter
  key? : ID
where
  key? elem dom counters
  counters(key?) = theta Counter
  counters' = counters ++ {key? |-> theta Counter'}
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{schema}{PromoteBump}
\Delta Bank \\
\Delta Counter \\
key? : ID
\where
key? \in \dom counters \\
counters~key? = \theta Counter \\
counters' = counters \oplus \{key? \mapsto \theta Counter'\}
\end{schema}
\end{exget}

% ── Z BINDINGS ──────────────────────────────────
\section{Z Bindings}

\begin{tabularx}{\columnwidth}{LR}
theta S                & $\theta S$ (binding of $S$ in scope; fuzz-typed) \\
theta S'               & $\theta S'$ (primed-component binding) \\
\{| name == e |\}      & $\lblot name == e \rblot$ (literal — fuzz-incompatible) \\
\{| a == e, b == f |\} & multiple components \\
\{| |\}                & empty binding \\
\end{tabularx}

\smallskip
\needspace{14\baselineskip}
\textbf{Multi-typed comprehension with binding:}

\begin{exwrite}
\begin{verbatim}
{ t : Track; a : Album |
  t.albumId = a.albumId .
  {| trackId == t.trackId |} }
\end{verbatim}
\end{exwrite}
\begin{exget}
\begin{center}
$\{~ t : Track; a : Album \mid t.albumId = a.albumId$\\
$\bullet~ \lblot trackId == t.trackId \rblot ~\}$
\end{center}
\end{exget}

{\scriptsize Use \texttt{;} to separate variable-type pairs inside \texttt{\{ \}}. Binding components are \textbf{comma}-separated.}

\smallskip
{\scriptsize\textbf{Fuzz note:} the \texttt{\{|\ldots|\}} literal binding is fuzz-incompatible — the engine emits it outside any Z paragraph so it renders without participating in fuzz type-checking. Use \kw{theta} when you need a fuzz-typed binding inside a schema.}

\end{multicols}

\begin{multicols}{2}
\raggedcolumns

% ── PROOFS ──────────────────────────────────────
\section{Proofs}

\needspace{14\baselineskip}
\textbf{INFRULE: — named inference rule display:}

\begin{exwrite}%
\begin{verbatim}
INFRULE:
premise_1
premise_2
---
conclusion [Rule name]
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\infer[\mbox{Rule name}]{conclusion}{premise_1 & premise_2}$
\end{center}
\end{exget}

{\scriptsize \kw{INFRULE:} renders a labelled rule schema. Use for displaying rule definitions (not proofs). The three-dash line \texttt{---} separates premises from conclusion.}

\smallskip
\needspace{4\baselineskip}
\textbf{How proof trees work (\kw{PROOF:}):}

{\scriptsize The source is \textbf{conclusion-first} (rendered tree reads
bottom-up). Premises are indented under their conclusion. A rule with
two or more premises requires \texttt{::} on each premise --- without
it, indented lines collapse into a \emph{linear chain} (each line
becomes the sole premise of the line above) and the extra premises
are silently dropped. Unary rules take a single indented child with
no \texttt{::}.}

\smallskip
\needspace{4\baselineskip}
\textbf{Proof tree syntax:}

\begin{tabularx}{\columnwidth}{LR}
\textrm{indent}          & Premises indented under conclusion \\
{[rule]}                  & Justification label \\
{[rule from $n$]}         & Discharge assumption $n$ \\
{[$n$] expr [assumption]} & Boxed assumption labelled $n$ \\
expr [from $n$]           & Leaf reference to assumption $n$ \\
::                        & Premise of a multi-premise rule \\
case $X$:                 & Branch in a \kw{lor elim} case analysis \\
\end{tabularx}

{\footnotesize\textbf{Rules:} \texttt{land intro}, \texttt{land elim 1/2}, \texttt{lor intro 1/2}, \texttt{lor elim}, \texttt{=> intro/elim}, \texttt{<=> intro/elim}, \texttt{lnot intro/elim}, \texttt{forall intro/elim}, \texttt{exists intro/elim}.}

\smallskip
\needspace{14\baselineskip}
\textbf{Modus ponens (binary --- requires \texttt{::} on each premise):}

\begin{exwrite}%
\begin{verbatim}
PROOF:
q [=> elim]
  :: p => q
  :: p
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\infer[\Rightarrow\textrm{-elim}]{q}{p \Rightarrow q & p}$
\end{center}
\end{exget}

\textbf{$\land$-intro (binary):}

\begin{exwrite}%
\begin{verbatim}
PROOF:
p land q [land intro]
  :: p
  :: q
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\infer[\land\textrm{-intro}]{p \land q}{p & q}$
\end{center}
\end{exget}

\textbf{$\Rightarrow$-intro with discharge (use \kw{from N}):}

\begin{exwrite}%
\begin{verbatim}
PROOF:
p => p [=> intro from 1]
  [1] p [assumption]
  :: p [from 1]
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\infer[\Rightarrow\textrm{-intro}^{[1]}]{p \Rightarrow p}{\ulcorner p \urcorner^{[1]}}$
\end{center}
\end{exget}

{\scriptsize \kw{[1] p [assumption]} declares the discharge binder.
\kw{p [from 1]} marks every leaf use. \kw{[=> intro from 1]} discharges it.}

\textbf{$\lor$-elim / case analysis (ternary --- disjunction + two cases):}

\begin{exwrite}%
\begin{verbatim}
PROOF:
(p lor q) => r [=> intro from 1]
  [1] p lor q [assumption]
  :: r [lor elim from 1]
    :: p lor q [from 1]
    case p:
      :: r [...derive r from p...]
    case q:
      :: r [...derive r from q...]
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{center}
$\infer[\lor\textrm{-elim}]{r}{p \lor q & \infer{r}{\ulcorner p \urcorner^{[1]}} & \infer{r}{\ulcorner q \urcorner^{[1]}}}$
\end{center}
\end{exget}

{\scriptsize \kw{lor elim from N} discharges the disjunction's assumption [N];
inside each \kw{case X:} block, the case formula is referenced as \kw{X [from N]}.}

\end{multicols}

\section{Relational Database Notation}

\needspace{14\baselineskip}
\textbf{Primary key annotation.}

{\scriptsize Mark a primary-key attribute with \kw{pk} in a named schema body. The annotation sits outside the Z box so fuzz type-checks cleanly.}

\begin{exwrite}
\begin{verbatim}
schema Book
  pk bookId : BookId
  isbn      : ISBN
  pages     : N
end
\end{verbatim}
\end{exwrite}
\begin{exget}
\begin{center}
\begin{schema}{Book}
bookId : BookId \\
isbn : ISBN \\
pages : \nat
\end{schema}\\[2pt]
\noindent$\mathrm{PK}(\mathrm{Book}) = \{bookId\}$
\end{center}
\end{exget}

{\scriptsize Composite: two \kw{pk} lines $\Rightarrow$ $\mathrm{PK}(S)=\{a,b\}$.
Comma-separated names on one \kw{pk} line also work.
\kw{pk} is rejected in \texttt{axdef}/\texttt{gendef}/inline schema text.}

\smallskip
\needspace{4\baselineskip}
\textbf{Relational algebra.}

\begin{tabularx}{\columnwidth}{LR}
sigma[p](R)          & $\mathrm{Restrict}_{p}(R)$ — restrict \\
pi[A,B](R)           & $\mathrm{Project}_{A,B}(R)$ — project \\
R[B/A]               & $R[B/A]$ — rename \\
R join S             & $\mathrm{Join}(R, S)$ — natural join \\
R join [p] S         & $\mathrm{Join}_{p}(R, S)$ — theta-join \\
R div S              & $R \div S$ — division \\
\end{tabularx}

{\scriptsize\textbf{Fuzz note:} relational-algebra operators are fuzz-incompatible — the engine emits them outside any Z paragraph so they render without participating in fuzz type-checking. Do not wrap algebra inside \texttt{zed}/\texttt{axdef}/\texttt{schema} blocks.}

\smallskip
\needspace{14\baselineskip}
\textbf{FK predicate form.}

{\scriptsize Foreign-key constraints are expressed as axdef predicates:}

\begin{exwrite}
\begin{verbatim}
axdef
where
  (R, S) |-> {a |-> b} elem FK
end
\end{verbatim}
\end{exwrite}
\begin{exget}%
\begin{axdef}
\where
(R, S) \mapsto \{a \mapsto b\} \in FK
\end{axdef}
\end{exget}

\smallskip
\needspace{4\baselineskip}
\textbf{GROUP \& UNGROUP.}

{\scriptsize Date's nested-relation operators.}

\begin{tabularx}{\columnwidth}{LR}
R group (\{A,B\} as x) & $R\ \mathop{\mathrm{GROUP}} (\{A,B\}\ \mathop{\mathrm{AS}}\ x)$ \\
R ungroup x          & $R\ \mathop{\mathrm{UNGROUP}}\ x$ \\
\end{tabularx}

\begin{exwrite}
\begin{verbatim}
Members group ({name,email} as contact)
Members ungroup contact
\end{verbatim}
\end{exwrite}
\begin{exget}
\begin{center}
$Members\ \mathop{\mathrm{GROUP}}\ (\{name,email\}\ \mathop{\mathrm{AS}}\ contact)$\\[3pt]
$Members\ \mathop{\mathrm{UNGROUP}}\ contact$
\end{center}
\end{exget}

{\scriptsize\textbf{Fuzz note:} GROUP and UNGROUP are fuzz-incompatible — the engine emits them outside any Z paragraph so they render without participating in fuzz type-checking. Do not wrap GROUP/UNGROUP inside \texttt{zed}/\texttt{axdef}/\texttt{schema} blocks.}

\smallskip
\textbf{GROUP aggregator form.}

{\scriptsize The aggregator keywords are \kw{Count}, \kw{Sum}, \kw{Avg}, \kw{Min}, \kw{Max}, \kw{Median}. Each takes one attribute name and an alias. The regroup form (\texttt{\{...\} as alias}) and the aggregator form are mutually exclusive in a single \texttt{group} expression.}

\begin{tabularx}{\columnwidth}{LR}
R group (Count(x) as n)   & $R \mathop{\mathrm{Group}}(\mathrm{Count}(x)~\mathrm{as}~n)$ \\
R group (Sum(x) as n)     & $R \mathop{\mathrm{Group}}(\mathrm{Sum}(x)~\mathrm{as}~n)$ \\
R group (Avg(x) as n)     & $R \mathop{\mathrm{Group}}(\mathrm{Avg}(x)~\mathrm{as}~n)$ \\
R group (Min(x) as n)     & $R \mathop{\mathrm{Group}}(\mathrm{Min}(x)~\mathrm{as}~n)$ \\
R group (Max(x) as n)     & $R \mathop{\mathrm{Group}}(\mathrm{Max}(x)~\mathrm{as}~n)$ \\
R group (Median(x) as n)  & $R \mathop{\mathrm{Group}}(\mathrm{Median}(x)~\mathrm{as}~n)$ \\
\end{tabularx}

\begin{exwrite}
\begin{verbatim}
Sales group (Count(orderId) as orderCount,
             Sum(amount) as totalRevenue)
\end{verbatim}
\end{exwrite}
\begin{exget}
\begin{center}
$Sales~\mathrm{Group}(\mathrm{Count}(orderId)~\mathrm{as}~orderCount,$\\
$\mathrm{Sum}(amount)~\mathrm{as}~totalRevenue)$
\end{center}
\end{exget}

{\scriptsize Multiple aggregators are comma-separated. Each accepts exactly one attribute identifier — no nested aggregators, no compound expressions.}
